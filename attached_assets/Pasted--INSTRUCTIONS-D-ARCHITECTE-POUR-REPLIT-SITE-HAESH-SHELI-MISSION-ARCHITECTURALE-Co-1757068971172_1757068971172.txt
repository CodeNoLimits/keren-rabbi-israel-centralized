# 🏗️ INSTRUCTIONS D'ARCHITECTE POUR REPLIT - SITE HAESH SHELI

## 🎯 **MISSION ARCHITECTURALE**

Construire le site **Haesh Sheli** de manière **absolument parfaite** en utilisant le code original comme base, mais avec des **améliorations majeures** en termes de :
- ✅ **Sections** : Structure moderne et optimisée
- ✅ **Fonctions** : E-commerce avancé et interactivité
- ✅ **Clarté** : Code propre et maintenable
- ✅ **Design** : Interface moderne et responsive

## 📋 **PHASE 1 : CONFIGURATION INITIALE REPLIT**

### **1.1 Création du Projet**
```bash
# Dans Replit, créer un nouveau projet "HTML/CSS/JS"
# Nom du projet : "haesh-sheli-premium"
```

### **1.2 Structure des Dossiers**
```
haesh-sheli-premium/
```

### **1.3 Installation des Dépendances**
```html
<!-- Dans index.html, ajouter ces CDN -->
<!-- Tailwind CSS pour le styling rapide -->
<script src="https://cdn.tailwindcss.com"></script>

<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.0/all.min.css">

<!-- Google Fonts pour Noto Sans Hebrew -->
<link href="https://fonts.googleapis.com/css2?family=Noto+Sans+Hebrew:wght@300;400;500;600;700;900&display=swap" rel="stylesheet">

<!-- AOS (Animate On Scroll) pour les animations -->
<link href="https://unpkg.com/aos@2.3.1/dist/aos.css" rel="stylesheet">
<script src="https://unpkg.com/aos@2.3.1/dist/aos.js"></script>

<!-- Swiper pour les carousels -->
<link rel="stylesheet" href="https://unpkg.com/swiper@8/swiper-bundle.min.css">
<script src="https://unpkg.com/swiper@8/swiper-bundle.min.js"></script>
```

```javascript
// ===== MAIN JAVASCRIPT - HAESH SHELI PREMIUM =====

class HaeshSheliApp {
    constructor() {
        this.init();
        this.bindEvents();
        this.initializeComponents();
    }

    init() {
        // Configuration
        this.config = {
            apiUrl: 'https://api.haesh-sheli.co.il',
            currency: '₪',
            language: 'he',
            rtl: true,
            animations: {
                duration: 300,
                easing: 'ease-in-out'
            }
        };

        // State
        this.state = {
            cart: JSON.parse(localStorage.getItem('haesh_sheli_cart')) || [],
            wishlist: JSON.parse(localStorage.getItem('haesh_sheli_wishlist')) || [],
            user: JSON.parse(localStorage.getItem('haesh_sheli_user')) || null,
            isLoading: false,
            mobileMenuOpen: false,
            cartOpen: false,
            searchOpen: false
        };

        // Elements
        this.elements = {
            body: document.body,
            header: document.getElementById('main-header'),
            mobileMenu: document.getElementById('mobile-menu'),
            cartSidebar: document.getElementById('cart-sidebar'),
            searchBar: document.getElementById('search-bar'),
            loadingScreen: document.getElementById('loading-screen')
        };

        console.log('🔥 Haesh Sheli App initialized');
    }

    bindEvents() {
        // Window events
        window.addEventListener('load', () => this.handlePageLoad());
        window.addEventListener('scroll', () => this.handleScroll());
        window.addEventListener('resize', () => this.handleResize());

        // Navigation events
        document.addEventListener('click', (e) => this.handleGlobalClick(e));
        document.addEventListener('keydown', (e) => this.handleKeydown(e));

        // Form events
        document.addEventListener('submit', (e) => this.handleFormSubmit(e));

        // Custom events
        document.addEventListener('cartUpdated', () => this.updateCartUI());
        document.addEventListener('wishlistUpdated', () => this.updateWishlistUI());
    }

    initializeComponents() {
        // Initialize AOS (Animate On Scroll)
        if (typeof AOS !== 'undefined') {
            AOS.init({
                duration: 800,
                easing: 'ease-in-out',
                once: true,
                offset: 100
            });
        }

        // Initialize Swiper carousels
        this.initializeCarousels();

        // Initialize lazy loading
        this.initializeLazyLoading();

        // Initialize tooltips
        this.initializeTooltips();

        // Update UI
        this.updateCartUI();
        this.updateWishlistUI();
    }

    // ===== PAGE LOAD HANDLING =====
    handlePageLoad() {
        // Hide loading screen
        setTimeout(() => {
            if (this.elements.loadingScreen) {
                this.elements.loadingScreen.classList.add('hidden');
                setTimeout(() => {
                    this.elements.loadingScreen.style.display = 'none';
                }, 500);
            }
        }, 1000);

        // Initialize page-specific functionality
        this.initializePageSpecific();

        // Trigger initial animations
        this.triggerInitialAnimations();
    }

    initializePageSpecific() {
        const page = document.body.dataset.page || 'home';
        
        switch (page) {
            case 'home':
                this.initializeHomePage();
                break;
            case 'shop':
                this.initializeShopPage();
                break;
            case 'product':
                this.initializeProductPage();
                break;
            default:
                console.log(`Page ${page} initialized`);
        }
    }

    initializeHomePage() {
        // Hero animations
        this.animateHeroElements();
        
        // Initialize book cards
        this.initializeBookCards();
        
        // Initialize testimonials carousel
        this.initializeTestimonialsCarousel();
    }

    animateHeroElements() {
        const heroElements = document.querySelectorAll('.hero-content [data-aos]');
        heroElements.forEach((el, index) => {
            setTimeout(() => {
                el.classList.add('animate-fade-in-up');
            }, index * 200);
        });
    }

    initializeBookCards() {
        const bookCards = document.querySelectorAll('.book-card');
        
        bookCards.forEach(card => {
            // Add to cart functionality
            const addToCartBtn = card.querySelector('.add-to-cart');
            if (addToCartBtn) {
                addToCartBtn.addEventListener('click', (e) => {
                    e.preventDefault();
                    const bookId = addToCartBtn.dataset.bookId;
                    this.addToCart(bookId);
                });
            }

            // Wishlist functionality
            const wishlistBtn = card.querySelector('.wishlist-btn');
            if (wishlistBtn) {
                wishlistBtn.addEventListener('click', (e) => {
                    e.preventDefault();
                    const bookId = card.dataset.bookId || addToCartBtn?.dataset.bookId;
                    this.toggleWishlist(bookId);
                });
            }

            // Quick view functionality
            const quickViewBtn = card.querySelector('.quick-view');
            if (quickViewBtn) {
                quickViewBtn.addEventListener('click', (e) => {
                    e.preventDefault();
                    const bookId = quickViewBtn.dataset.bookId;
                    this.openQuickView(bookId);
                });
            }

            // Hover effects
            card.addEventListener('mouseenter', () => {
                card.style.transform = 'translateY(-8px)';
            });

            card.addEventListener('mouseleave', () => {
                card.style.transform = 'translateY(0)';
            });
        });
    }

    // ===== SCROLL HANDLING =====
    handleScroll() {
        const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
        
        // Header scroll effect
        if (this.elements.header) {
            if (scrollTop > 100) {
                this.elements.header.classList.add('scrolled');
            } else {
                this.elements.header.classList.remove('scrolled');
            }
        }

        // Parallax effects
        this.handleParallax(scrollTop);

        // Progress indicator
        this.updateProgressIndicator(scrollTop);
    }

    handleParallax(scrollTop) {
        const parallaxElements = document.querySelectorAll('[data-parallax]');
        
        parallaxElements.forEach(el => {
            const speed = parseFloat(el.dataset.parallax) || 0.5;
            const yPos = -(scrollTop * speed);
            el.style.transform = `translateY(${yPos}px)`;
        });
    }

    updateProgressIndicator(scrollTop) {
        const progressBar = document.querySelector('.progress-indicator');
        if (!progressBar) return;

        const windowHeight = window.innerHeight;
        const documentHeight = document.documentElement.scrollHeight;
        const progress = (scrollTop / (documentHeight - windowHeight)) * 100;
        
        progressBar.style.width = `${Math.min(progress, 100)}%`;
    }

    // ===== GLOBAL CLICK HANDLING =====
    handleGlobalClick(e) {
        const target = e.target;
        const closest = target.closest.bind(target);

        // Mobile menu toggle
        if (closest('.mobile-menu-toggle')) {
            e.preventDefault();
            this.toggleMobileMenu();
        }

        // Cart toggle
        if (closest('.cart-toggle')) {
            e.preventDefault();
            this.toggleCart();
        }

        // Search toggle
        if (closest('.search-toggle')) {
            e.preventDefault();
            this.toggleSearch();
        }

        // Wishlist toggle
        if (closest('.wishlist-toggle')) {
            e.preventDefault();
            this.toggleWishlistSidebar();
        }

        // Close menus when clicking outside
        if (!closest('.mobile-menu') && !closest('.mobile-menu-toggle')) {
            this.closeMobileMenu();
        }

        if (!closest('.cart-sidebar') && !closest('.cart-toggle')) {
            this.closeCart();
        }

        // Smooth scroll for anchor links
        if (target.matches('a[href^="#"]')) {
            e.preventDefault();
            const targetId = target.getAttribute('href');
            const targetElement = document.querySelector(targetId);
            
            if (targetElement) {
                this.smoothScrollTo(targetElement);
            }
        }
    }

    // ===== KEYBOARD HANDLING =====
    handleKeydown(e) {
        switch (e.key) {
            case 'Escape':
                this.closeAllMenus();
                break;
            case 'Enter':
                if (e.target.matches('.search-input')) {
                    this.performSearch(e.target.value);
                }
                break;
        }
    }

    // ===== MENU FUNCTIONS =====
    toggleMobileMenu() {
        this.state.mobileMenuOpen = !this.state.mobileMenuOpen;
        
        if (this.state.mobileMenuOpen) {
            this.openMobileMenu();
        } else {
            this.closeMobileMenu();
        }
    }

    openMobileMenu() {
        if (this.elements.mobileMenu) {
            this.elements.mobileMenu.classList.add('active');
            this.elements.body.classList.add('menu-open');
            this.state.mobileMenuOpen = true;
        }
    }

    closeMobileMenu() {
        if (this.elements.mobileMenu) {
            this.elements.mobileMenu.classList.remove('active');
            this.elements.body.classList.remove('menu-open');
            this.state.mobileMenuOpen = false;
        }
    }

    toggleCart() {
        this.state.cartOpen = !this.state.cartOpen;
        
        if (this.state.cartOpen) {
            this.openCart();
        } else {
            this.closeCart();
        }
    }

    openCart() {
        if (this.elements.cartSidebar) {
            this.elements.cartSidebar.classList.add('active');
            this.elements.body.classList.add('cart-open');
            this.state.cartOpen = true;
            this.updateCartContent();
        }
    }

    closeCart() {
        if (this.elements.cartSidebar) {
            this.elements.cartSidebar.classList.remove('active');
            this.elements.body.classList.remove('cart-open');
            this.state.cartOpen = false;
        }
    }

    toggleSearch() {
        this.state.searchOpen = !this.state.searchOpen;
        
        if (this.elements.searchBar) {
            if (this.state.searchOpen) {
                this.elements.searchBar.classList.remove('hidden');
                const searchInput = this.elements.searchBar.querySelector('input');
                if (searchInput) {
                    setTimeout(() => searchInput.focus(), 100);
                }
            } else {
                this.elements.searchBar.classList.add('hidden');
            }
        }
    }

    closeAllMenus() {
        this.closeMobileMenu();
        this.closeCart();
        this.state.searchOpen = false;
        if (this.elements.searchBar) {
            this.elements.searchBar.classList.add('hidden');
        }
    }

    // ===== CART FUNCTIONS =====
    addToCart(bookId, quantity = 1) {
        const book = this.getBookData(bookId);
        if (!book) {
            this.showNotification('ספר לא נמצא', 'error');
            return;
        }

        const existingItem = this.state.cart.find(item => item.id === bookId);
        
        if (existingItem) {
            existingItem.quantity += quantity;
        } else {
            this.state.cart.push({
                id: bookId,
                ...book,
                quantity: quantity,
                addedAt: new Date().toISOString()
            });
        }

        this.saveCart();
        this.updateCartUI();
        this.showNotification(`${book.title} נוסף לעגלה`, 'success');
        
        // Trigger custom event
        document.dispatchEvent(new CustomEvent('cartUpdated'));
    }

    removeFromCart(bookId) {
        this.state.cart = this.state.cart.filter(item => item.id !== bookId);
        this.saveCart();
        this.updateCartUI();
        document.dispatchEvent(new CustomEvent('cartUpdated'));
    }

    updateCartQuantity(bookId, quantity) {
        const item = this.state.cart.find(item => item.id === bookId);
        if (item) {
            if (quantity <= 0) {
                this.removeFromCart(bookId);
            } else {
                item.quantity = quantity;
                this.saveCart();
                this.updateCartUI();
                document.dispatchEvent(new CustomEvent('cartUpdated'));
            }
        }
    }

    clearCart() {
        this.state.cart = [];
        this.saveCart();
        this.updateCartUI();
        this.showNotification('העגלה נוקתה', 'info');
        document.dispatchEvent(new CustomEvent('cartUpdated'));
    }

    getCartTotal() {
        return this.state.cart.reduce((total, item) => {
            return total + (item.price * item.quantity);
        }, 0);
    }

    getCartItemCount() {
        return this.state.cart.reduce((total, item) => {
            return total + item.quantity;
        }, 0);
    }

    saveCart() {
        localStorage.setItem('haesh_sheli_cart', JSON.stringify(this.state.cart));
    }

    updateCartUI() {
        // Update cart count
        const cartCounts = document.querySelectorAll('.cart-count');
        const itemCount = this.getCartItemCount();
        
        cartCounts.forEach(count => {
            count.textContent = itemCount;
            count.style.display = itemCount > 0 ? 'flex' : 'none';
        });

        // Update cart total
        const cartTotals = document.querySelectorAll('.cart-total');
        const total = this.getCartTotal();
        
        cartTotals.forEach(totalEl => {
            totalEl.textContent = `${total.toFixed(2)} ${this.config.currency}`;
        });

        // Update cart content
        this.updateCartContent();
    }

    updateCartContent() {
        const cartContent = document.querySelector('.cart-content');
        if (!cartContent) return;

        if (this.state.cart.length === 0) {
            cartContent.innerHTML = `
                <div class="empty-cart text-center py-12">
                    <i class="fas fa-shopping-cart text-6xl text-gray-300 mb-4"></i>

                    <p class="text-gray-500 mb-6">הוסף ספרים לעגלה כדי להמשיך</p>
                    <a href="#featured-books" class="btn btn-primary">התחל קנייה</a>
                </div>
            `;
        } else {
            const cartItems = this.state.cart.map(item => `
                <div class="cart-item flex items-center gap-4 p-4 border-b border-gray-200">
                    <img src="${item.image}" alt="${item.title}" class="w-16 h-16 object-cover rounded-lg">
                    <div class="flex-1">
                        <h4 class="font-semibold text-gray-800">${item.title}</h4>
                        <p class="text-sm text-gray-600">${item.author}</p>
                        <div class="flex items-center gap-2 mt-2">
                            <button class="quantity-btn minus" data-book-id="${item.id}">
                                <i class="fas fa-minus"></i>
                            </button>
                            <span class="quantity px-3 py-1 bg-gray-100 rounded">${item.quantity}</span>
                            <button class="quantity-btn plus" data-book-id="${item.id}">
                                <i class="fas fa-plus"></i>
                            </button>
                        </div>
                    </div>
                    <div class="text-left">
                        <div class="font-semibold text-primary">${(item.price * item.quantity).toFixed(2)} ${this.config.currency}</div>
                        <button class="remove-item text-red-500 hover:text-red-700 text-sm mt-1" data-book-id="${item.id}">
                            <i class="fas fa-trash ml-1"></i>
                            הסר
                        </button>
                    </div>
                </div>
            `).join('');

            const total = this.getCartTotal();
            
            cartContent.innerHTML = `
                <div class="cart-items">
                    ${cartItems}
                </div>
                <div class="cart-summary p-4 bg-gray-50">
                    <div class="flex justify-between items-center mb-4">
                        <span class="text-lg font-semibold">סה"כ:</span>
                        <span class="text-2xl font-bold text-primary">${total.toFixed(2)} ${this.config.currency}</span>
                    </div>
                    <div class="flex gap-2">
                        <button class="btn btn-outline flex-1" onclick="app.clearCart()">נקה עגלה</button>
                        <button class="btn btn-primary flex-1" onclick="app.proceedToCheckout()">המשך לתשלום</button>
                    </div>
                </div>
            `;

            // Bind quantity change events
            cartContent.querySelectorAll('.quantity-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const bookId = btn.dataset.bookId;
                    const item = this.state.cart.find(item => item.id === bookId);
                    if (item) {
                        const newQuantity = btn.classList.contains('plus') 
                            ? item.quantity + 1 
                            : item.quantity - 1;
                        this.updateCartQuantity(bookId, newQuantity);
                    }
                });
            });

            // Bind remove item events
            cartContent.querySelectorAll('.remove-item').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const bookId = btn.dataset.bookId;
                    this.removeFromCart(bookId);
                });
            });
        }
    }

    toggleWishlist(bookId) {
        const index = this.state.wishlist.indexOf(bookId);
        
        if (index > -1) {
            this.state.wishlist.splice(index, 1);
            this.showNotification('הוסר מרשימת המשאלות', 'info');
        } else {
            this.state.wishlist.push(bookId);
            this.showNotification('נוסף לרשימת המשאלות', 'success');
        }

        this.saveWishlist();
        this.updateWishlistUI();
        document.dispatchEvent(new CustomEvent('wishlistUpdated'));
    }

    saveWishlist() {
        localStorage.setItem('haesh_sheli_wishlist', JSON.stringify(this.state.wishlist));
    }

    updateWishlistUI() {
        // Update wishlist count
        const wishlistCounts = document.querySelectorAll('.wishlist-count');
        const count = this.state.wishlist.length;
        
        wishlistCounts.forEach(countEl => {
            countEl.textContent = count;
            countEl.style.display = count > 0 ? 'flex' : 'none';
        });

        // Update wishlist buttons
        const wishlistBtns = document.querySelectorAll('.wishlist-btn');
        wishlistBtns.forEach(btn => {
            const bookId = btn.dataset.bookId || btn.closest('.book-card')?.dataset.bookId;
            if (bookId && this.state.wishlist.includes(bookId)) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
        });
    }

            error: 'bg-red-500 text-white',

            info: 'bg-blue-500 text-white'
        };

    initializeLazyLoading() {
        const images = document.querySelectorAll('img[data-src]');
        
        const imageObserver = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const img = entry.target;
                    img.src = img.dataset.src;
                    img.classList.remove('lazy');
                    imageObserver.unobserve(img);
                }
            });
        });

        images.forEach(img => imageObserver.observe(img));
    }

        const tooltip = document.createElement('div');
        tooltip.className = 'tooltip absolute z-50 bg-gray-900 text-white px-3 py-2 rounded-lg text-sm';
        tooltip.textContent = text;
        tooltip.id = 'active-tooltip';
        
        document.body.appendChild(tooltip);
        
        const rect = element.getBoundingClientRect();
        tooltip.style.top = `${rect.top - tooltip.offsetHeight - 8}px`;
        tooltip.style.left = `${rect.left + (rect.width - tooltip.offsetWidth) / 2}px`;
    }

    hideTooltip() {
        const tooltip = document.getElementById('active-tooltip');
        if (tooltip) {
            document.body.removeChild(tooltip);
```
